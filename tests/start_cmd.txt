# 使用完整参数名 --config-file
cargo run --bin edgion-controller -- --config-file ./config/edgion-controller.toml
cargo run --bin edgion-gateway -- --config-file ./config/edgion-gateway.toml

# 或使用短参数 -c
cargo run --bin edgion-controller -- -c ./config/edgion-controller.toml
cargo run --bin edgion-gateway -- -c ./config/edgion-gateway.toml
